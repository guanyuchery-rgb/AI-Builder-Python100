## 今日最小可运行任务

```python
import time

cache = {}

def slow_double(x: int) -> int:
    if x in cache:
        return cache[x]
    time.sleep(0.1)
    cache[x] = x * 2
    return cache[x]

start = time.perf_counter()
print(slow_double(3), slow_double(3))
print("seconds", round(time.perf_counter() - start, 3))
```

```python
import pandas as pd

df = pd.DataFrame({"ret": [0.01, -0.02, 0.03], "position": [1, 0, 1]})
df["strategy_ret"] = df["position"] * df["ret"]
df["equity"] = (1 + df["strategy_ret"]).cumprod()
print(df)
```
